itk_wrap_module("MediPyGeometry")
    set(WRAPPER_LIBRARY_DEPENDS ITKCommon ITKStatistics)
    
    itk_wrap_submodule("itkPlane")
        itk_wrap_class("itk::Plane")
            foreach(type "F" "D")
                itk_wrap_template(${type} ${ITKT_${type}})
            endforeach()
        itk_end_wrap_class()
    itk_end_wrap_submodule()
    
    itk_wrap_submodule("itkSymmetryPlaneCalculator")
        itk_wrap_class("itk::SymmetryPlaneCalculator" POINTER)
            itk_wrap_image_filter("${WRAP_ITK_REAL}" 1 3)
        itk_end_wrap_class()
    itk_end_wrap_submodule()
itk_end_wrap_module()
